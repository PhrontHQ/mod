- function numberToBase(n, b) {
    if (n === 0n) {
        return [0n];
    }
    let digits = [];
    while (n > 0n) {
        digits.push(n % b);
        n = n / b;
    }
    return digits.reverse();
}

BigInt Literals: Use 0n for zero and ensure all arithmetic operations involve BigInt.
